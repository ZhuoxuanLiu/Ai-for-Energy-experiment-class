class MyLogisticRegression:
    def __init__(self, learning_rate=0.01, num_iterations=1000):
        self.learning_rate = learning_rate
        self.num_iterations = num_iterations
        self.weights = None
        self.bias = None

    def sigmoid(self, z):
        return 1 / (1 + np.exp(-z))

    def fit(self, X, y):
        # 初始化模型参数
        num_samples, num_features = X.shape
        self.weights = np.zeros(num_features)
        num_classes = len(np.unique(y))
        self.bias = 0

        self.models = []

        # 训练多个二分类逻辑回归模型
        for i in range(num_classes):
            # 创建一个类别的二分类目标向量
            y_binary = np.where(y == i, 1, 0)

            # 初始化模型参数
            weights = np.zeros(num_features)
            bias = 0

            # 梯度下降训练
            for _ in range(self.num_iterations):
                # 前向传播
                linear_model = np.dot(X, weights) + bias
                y_pred = self.sigmoid(linear_model)

                # 计算梯度
                dw = (1 / num_samples) * np.dot(X.T, (y_pred - y_binary))
                db = (1 / num_samples) * np.sum(y_pred - y_binary)

                # 更新参数
                weights -= self.learning_rate * dw
                bias -= self.learning_rate * db

            # 将训练好的模型保存到列表中
            self.models.append((weights, bias))

    def predict(self, X):
        num_samples = X.shape[0]
        num_classes = len(self.models)
        predictions = np.zeros((num_samples, num_classes))

        # 对每个模型进行预测
        for i, (weights, bias) in enumerate(self.models):
            linear_model = np.dot(X, weights) + bias
            y_pred = self.sigmoid(linear_model)
            predictions[:, i] = y_pred

        # 选择具有最大预测概率的类别作为最终预测结果
        y_pred_class = np.argmax(predictions, axis=1)
        return y_pred_class

# 初始化多分类模型（这里使用逻辑回归作为基分类器）
classifier = OneVsRestClassifier(LogisticRegression())

# 拟合模型
classifier.fit(X_train_std, y_train_enc)

# 预测概率
y_score = classifier.predict_proba(X_test_std)

# 计算每个类别的精确度和召回率
precision = dict()
recall = dict()
average_precision = dict()
# precision = TP / (TP + FP)  预测真实正例/预测正例
# recall = TP / (TP + FN)  预测真实正例/真实正例

for i in range(len(classifier.classes_)):
    # note that precision and recall return arrays of shape (n_thresholds + 1,)
    # n_thresholds = len(np.unique(probas_pred))
    precision[i], recall[i], _ = precision_recall_curve(y_test_enc == i, y_score[:, i])
    average_precision[i] = round(average_precision_score(y_test_enc == i, y_score[:, i]), 2)

# 绘制每个类别的精确度-召回率曲线
plt.figure()
plt.step(recall[0], precision[0], where='post', label=f'Class 0 auc: {average_precision[0]}')
plt.step(recall[1], precision[1], where='post', label=f'Class 1 auc: {average_precision[1]}')
plt.step(recall[2], precision[2], where='post', label=f'Class 2 auc: {average_precision[2]}')
plt.step(recall[3], precision[3], where='post', label=f'Class 3 auc: {average_precision[3]}')
# ... 绘制其他类别的曲线
plt.xlabel('Recall')
plt.ylabel('Precision')
plt.title('Precision-Recall Curve')
plt.legend()
plt.show()

# precision -> 1, recall -> 0  所有都预测为负例
# precision -> 0, recall -> 1  所有都预测为正例
