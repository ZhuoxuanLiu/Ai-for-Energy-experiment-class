Gradient descent is an algorithm to solve the unconstrained optimization problem.¶

Given an unconstrained optimization problem $$ \underset{x}{min} \ f(x)$$

Where $x$ is the decision variable, and the $ f(x) $ is the objective function or optimization funtion.

We find the optimal value of $x$ in a iterative manner

$$ x{\leftarrow}x-{\eta}{ f'(x)} $$

The insight is that the gradient is a direction, if we move toward this direction, the value of function will increase.

class DerivativeCalculator:
    def __init__(self, func):
        self.func = func

    def grad(self, params: np.ndarray):
        return np.array([params[0], 4*(params[1]-2)])
    
    def loss(self, params: np.ndarray):
        return self.func(params)

class GradientDescendOptimizer:
    def __init__(self, params, grad_calculator, learning_rate):
        self.params = params
        self.grad_calculator = grad_calculator
        self.lr = learning_rate
        self.grad = None

    def step(self):
        grads = self.grad_calculator.grad(self.params)
        self.params = self.params - grads*0.01
        self.grad = grads

    def zero_grad(self):
        self.grad = None

class MomentumOptimizer:
    def __init__(self, params, grad_calculator, learning_rate, beta):
        self.params = params
        self.grad_calculator = grad_calculator
        self.lr = learning_rate
        self.beta = beta
        self.momentum = np.zeros_like(params)
        self.grad = None

    def step(self):
        grads = self.grad_calculator.grad(self.params)
        self.momentum = self.beta * self.momentum - self.lr * grads
        self.params += self.momentum
        self.grad = grads

    def zero_grad(self):
        self.momentum = np.zeros_like(self.params)
        self.grad = None
